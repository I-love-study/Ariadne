    class BotJoinGroupEvent(GroupEvent, BotEvent):

        当监听该事件或该类事件时, 请优先考虑使用原始事件类作为类型注解, 以此获得事件类实例, 便于获取更多的信息!

                if generic_issubclass(Group, interface.annotation):
